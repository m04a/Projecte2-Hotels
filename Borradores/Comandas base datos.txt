create database hotel;

create table usuario (
	usuari varchar(69) primary key,
	password varchar(69) not null,
	nombre varchar(69),
	apellidos varchar(69),
	fechanacimiento date,
	sexo tinyint(1),
	tipo varchar(69) default 'cliente'
);


create table hotel (
     nombre varchar(69) primary key,
     iniciovac date,
     finvac date,
     totalhab int(5)
);	

create table tipo (
	idtipo int(15) primary key auto_increment,
	precio int(10),
	imagen varchar(30),
	m2 int(5),
	cantidad int(5),
	persmax int(5),
	idhabitacion int(10),
	descripcion varchar(420),
	nom varchar(30),
	nomhotel varchar(69),
	foreign key tipo(nomhotel) references hotel(nombre)
);
	
create table reserva (
    numres int(10) primary key auto_increment,
    numpers int(5),
    finicio date,
    ffin date,
	idhabitacion int(10),
	idtipo  int(15),
	usuario varchar(69),
	foreign key reservatipo(idtipo) references tipo(idtipo),
	foreign key reservausuari(usuario) references usuario(usuari)
);
